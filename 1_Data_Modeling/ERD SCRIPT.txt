Locations
--
id PK int
city varchar(50)
country varchar(50)

MemberTypes
--
id PK int
member_type varchar(50)
location_id int fk >- Locations.id

Stations
--
station_id PK int
station_name varchar(255)
latitude numeric(11,8)
longitude numeric(11,8)
location_id int fk >- Locations.id

Bikes
--
id PK int
bike_id int
location_id int fk >- Locations.id

Ridership
--
id PK int
trip_id int
start_station_id int fk >- Stations.station_id
end_station_id int fk >- Stations.station_id
start_date datetime
end_date datetime
bike_id int fk >- Bikes.id
member_type_id int fk >- MemberTypes.id
location_id int

Pricing
--
price_id PK int
location_id int fk >- Locations.id
membertype_id int fk - MemberTypes.id
plan varchar(50)
amount numeric(10,2)
